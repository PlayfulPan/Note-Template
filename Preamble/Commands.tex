    {
        \IfNoValueTF{#2}
        {
            \mathrm{d}\indices{^{\ast}}#3
        }
        {
            \mathrm{d}\indices*{^{\ast}_{\scriptscriptstyle (#2)}}#3
        }
    }
    {
        \IfNoValueTF{#2}
        {
            \mathrm{d}#3
        }
        {
            \mathrm{d}\indices{_{\scriptscriptstyle (#2)}}#3
        }
    }
}

            D\indices{_{\!#3}}
        }
        {
            D\indices*{^{\scriptscriptstyle(#2)}_{\!#3}}
        }
    }
    {
        \IfNoValueTF{#2}
        {
            \nabla\indices{_{\!#3}}
        }
        {
            \nabla\indices*{^{\scriptscriptstyle(#2)}_{\!#3}}
        }
    }
}
